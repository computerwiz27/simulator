MVI reg0 0          ;;reg0 keeps track of iteration
MVI reg1 0          ;;load the first 2 elements of the fibonacci
MVI reg2 1          ;;sequence into reg1 and reg2 respectively
MVI reg10 10        ;;reg10 is the number of iterations n
BEQ reg10 reg0 11   ;;if we are on iteration n + 1 exit the program
WRT reg0 reg1       ;;write the contents of reg1 to mem addr in reg0
ADDI reg0 reg0 1    ;;increment reg0
ADD reg3 reg1 reg2  ;;calculate next element
MV reg1 reg2
MV reg2 reg3
JMP 4               ;;go to begining of loop
